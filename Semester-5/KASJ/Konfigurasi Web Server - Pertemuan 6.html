Apache Web Server Installation

Ubuntu/Debian Installation

# Update package repository
sudo apt update

# Install Apache2
sudo apt install apache2

# Enable and start Apache
sudo systemctl enable apache2
sudo systemctl start apache2

# Check status
sudo systemctl status apache2
Default Document Root:
/var/www/html/

CentOS/RHEL Installation

# Install Apache (httpd)
sudo yum install httpd
# or for newer versions
sudo dnf install httpd

# Enable and start service
sudo systemctl enable httpd
sudo systemctl start httpd

# Configure firewall
sudo firewall-cmd --permanent --add-service=http
sudo firewall-cmd --permanent --add-service=https
sudo firewall-cmd --reload
Config Files:
• /etc/apache2/ (Debian)
• /etc/httpd/ (RHEL)

Ubuntu/Debian Installation

# Update package repository

Apache Virtual Hosts Configuration
Hosting Multiple Websites

Create Virtual Host

1
Create Directory
sudo mkdir -p /var/www/example.com/html
2
Set Permissions
sudo chown -R www-data:www-data /var/www/example.com
3
Create Config File
/etc/apache2/sites-available/example.com.conf
4
Enable Site
sudo a2ensite example.com.conf

Virtual Host Configuration

<VirtualHost *:80>
    ServerName example.com
    ServerAlias www.example.com
    DocumentRoot /var/www/example.com/html
    
    ErrorLog ${APACHE_LOG_DIR}/example.com_error.log
    CustomLog ${APACHE_LOG_DIR}/example.com_access.log combined
    
    <Directory /var/www/example.com/html>
        Options Indexes FollowSymLinks
        AllowOverride All
        Require all granted
    </Directory>
</VirtualHost>
# Reload Apache configuration
sudo systemctl reload apache2

# Test configuration
sudo apache2ctl configtest
NGINX Server Blocks Configuration
NGINX Virtual Hosting

Setup Steps

1
Create Directory
sudo mkdir -p /var/www/example.com/html
2
Set Ownership
sudo chown -R nginx:nginx /var/www/example.com
3
Create Server Block
/etc/nginx/sites-available/example.com
4
Enable Site
sudo ln -s /etc/nginx/sites-available/example.com /etc/nginx/sites-enabled/

Server Block Configuration

server {
    listen 80;
    listen [::]:80;
    
    server_name example.com www.example.com;
    root /var/www/example.com/html;
    index index.html index.htm index.nginx-debian.html;
    
    location / {
        try_files $uri $uri/ =404;
    }
    
    access_log /var/log/nginx/example.com_access.log;
    error_log /var/log/nginx/example.com_error.log;
}
# Test configuration
sudo nginx -t

# Reload NGINX
sudo systemctl reload nginx

Let's Encrypt with Certbot
Free SSL Certificate Automation

Certbot Installation

# Ubuntu/Debian
sudo apt update
sudo apt install certbot

# For Apache
sudo apt install python3-certbot-apache

# For NGINX
sudo apt install python3-certbot-nginx

# CentOS/RHEL
sudo dnf install certbot python3-certbot-apache
Automatic Renewal:
Certificates renew automatically every 90 days

Certificate Generation

# Apache - Automatic configuration
sudo certbot --apache -d example.com -d www.example.com

# NGINX - Automatic configuration
sudo certbot --nginx -d example.com -d www.example.com

# Manual certificate generation
sudo certbot certonly --webroot -w /var/www/example.com/html -d example.com

# Test automatic renewal
sudo certbot renew --dry-run

# List certificates
sudo certbot certificates
Rate Limits:
50 certificates per domain per week
Manual SSL Configuration

Apache SSL Virtual Host

<VirtualHost *:443>
    ServerName example.com
    DocumentRoot /var/www/example.com/html
    
    SSLEngine on
    SSLCertificateFile /etc/ssl/certs/example.com.crt
    SSLCertificateKeyFile /etc/ssl/private/example.com.key
    SSLCertificateChainFile /etc/ssl/certs/ca-bundle.crt
    
    SSLProtocol all -SSLv2 -SSLv3
    SSLCipherSuite ECDHE-RSA-AES256-GCM-SHA384:ECDHE-RSA-AES128-GCM-SHA256
    SSLHonorCipherOrder on
</VirtualHost>

# Redirect HTTP to HTTPS
<VirtualHost *:80>
    ServerName example.com
    Redirect permanent / https://example.com/
</VirtualHost>

NGINX SSL Server Block

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name example.com;
    root /var/www/example.com/html;
    
    ssl_certificate /etc/ssl/certs/example.com.crt;
    ssl_certificate_key /etc/ssl/private/example.com.key;
    
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-RSA-AES256-GCM-SHA384:ECDHE-RSA-AES128-GCM-SHA256;
    ssl_prefer_server_ciphers off;
    
    add_header Strict-Transport-Security "max-age=63072000" always;
}

# Redirect HTTP to HTTPS
server {
    listen 80;
    server_name example.com;
    return 301 https://$server_name$request_uri;
}
Web Server Security Hardening

Server Headers

Security Headers 
# Apache (.htaccess)
Header always set X-Frame-Options DENY
Header always set X-Content-Type-Options nosniff
Header always set X-XSS-Protection "1; mode=block"
Header always set Referrer-Policy "strict-origin-when-cross-origin"

# NGINX
add_header X-Frame-Options DENY;
add_header X-Content-Type-Options nosniff;
add_header X-XSS-Protection "1; mode=block";

Hide Server Info

Information Disclosure 
# Apache
ServerTokens Prod
ServerSignature Off

# NGINX
server_tokens off;

# Remove Server header completely
more_clear_headers Server;

Access Control

File Permissions 
# Set proper permissions
sudo chmod -R 644 /var/www/html/
sudo chmod -R 755 /var/www/html/
sudo chown -R www-data:www-data /var/www/

# Disable directory browsing
Options -Indexes

# Block access to sensitive files
<Files ~ "^\.ht">
    Require all denied
</Files>
Critical Security Measures:
• Regular software updates • Strong SSL configuration • Firewall rules • Access log monitoring • Backup strategy • Intrusion detection system
Performance Tuning Basics

Apache Performance

Compression

# Enable mod_deflate
a2enmod deflate

# Compress text files
<Location "/">
    SetOutputFilter DEFLATE
    SetEnvIfNoCase Request_URI \
        \.(?:gif|jpe?g|png)$ no-gzip
</Location>

Caching

# Enable mod_expires
a2enmod expires

# Cache static files
<IfModule mod_expires.c>
    ExpiresActive On
    ExpiresByType image/jpg "access plus 1 month"
    ExpiresByType text/css "access plus 1 week"
</IfModule>

NGINX Performance

Worker Optimization

# nginx.conf
worker_processes auto;
worker_connections 1024;

events {
    use epoll;
    multi_accept on;
}

Gzip Compression

# Gzip compression
gzip on;
gzip_vary on;
gzip_min_length 1024;
gzip_types
    text/plain
    text/css
    application/json
    application/javascript;

Common Issues & Solutions
Issue Symptoms Solution
403 Forbidden Permission denied errors Check file permissions, directory permissions, SELinux
SSL Certificate Error Browser security warnings Verify certificate chain, check expiry, restart web server
Virtual Host Not Working Wrong site loads or 404 error Check DNS, verify virtual host config, enable site
High CPU Usage Server becomes unresponsive Check processes, optimize configuration, add caching
Configuration Syntax Error Server won't start or reload Run config test, check syntax, review recent changes
Port Already in Use Server binding fails Check running processes, kill conflicting services
